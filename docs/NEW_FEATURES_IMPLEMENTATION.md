            self.notify_callbacks("program_stopped")

    def notify_callbacks(self, event: str):
        """–£–≤–µ–¥–æ–º–ª–µ–Ω–∏–µ callback'–æ–≤ –æ–± –∏–∑–º–µ–Ω–µ–Ω–∏–∏ —Å–æ—Å—Ç–æ—è–Ω–∏—è"""
